19/// Anti-replay sliding window.
20///
21/// Per 1-SECURITY.md:
22/// - Window size: 2048 bits minimum
23/// - Below window: MUST reject
24/// - Seen nonce: MUST reject
25/// - Above highest: Update window
26pub struct ReplayWindow {
27    /// Bitmap for tracking seen nonces
28    bitmap: [u64; REPLAY_WINDOW_SIZE / 64],
29    /// Highest nonce seen so far
30    highest: u64,
31    /// Whether we've seen any packets yet
32    initialized: bool,
33}
34
35impl ReplayWindow {
36    /// Create a new replay window.
37    pub fn new() -> Self {
38        Self {
39            bitmap: [0; REPLAY_WINDOW_SIZE / 64],
40            highest: 0,
41            initialized: false,
42        }
43    }
44
45    /// Check if a nonce is a replay (without updating).
46    pub fn is_replay(&self, nonce: u64) -> bool {
47        if !self.initialized {
48            return false;
49        }
50
51        if nonce > self.highest {
52            return false;
53        }
54
55        let diff = self.highest - nonce;
56        if diff >= REPLAY_WINDOW_SIZE as u64 {
57            return true; // Below window
58        }
59
60        let bit_index = diff as usize;
61        let word_index = bit_index / 64;
62        let bit_offset = bit_index % 64;
63        (self.bitmap[word_index] & (1 << bit_offset)) != 0
64    }
65
66    /// Check if a nonce is a replay and update the window.
67    ///
68    /// Returns Ok(()) if the nonce is valid (not seen before).
69    /// Returns Err(ReplayDetected) if the nonce is a replay.
70    ///
71    /// Per 1-SECURITY.md, replay check MUST occur BEFORE AEAD verification.
72    pub fn check_and_update(&mut self, nonce: u64) -> Result<(), CryptoError> {
73        if !self.initialized {
74            // First packet - initialize
75            self.highest = nonce;
76            self.mark_seen(nonce);
77            self.initialized = true;
78            return Ok(());
79        }
80
81        if nonce > self.highest {
82            // Advance the window
83            let shift = nonce - self.highest;
84            self.shift_window(shift);
85            self.highest = nonce;
86            self.mark_seen(nonce);
87            Ok(())
88        } else {
89            let diff = self.highest - nonce;
90            if diff >= REPLAY_WINDOW_SIZE as u64 {
91                // Too old - below window
92                return Err(CryptoError::ReplayDetected);
93            }
94
95            // Check if already seen
96            if self.is_seen(nonce) {
97                return Err(CryptoError::ReplayDetected);
98            }
99
100            // Mark as seen
101            self.mark_seen(nonce);
102            Ok(())
103        }
104    }
105
106    /// Check if a nonce has been seen (internal helper).
107    fn is_seen(&self, nonce: u64) -> bool {
108        if nonce > self.highest {
109            return false;
110        }
111        let diff = self.highest - nonce;
112        if diff >= REPLAY_WINDOW_SIZE as u64 {
113            return true; // Treat below-window as "seen" (rejected)
114        }
115        let bit_index = diff as usize;
116        let word_index = bit_index / 64;
117        let bit_offset = bit_index % 64;
118        (self.bitmap[word_index] & (1 << bit_offset)) != 0
119    }
120
121    /// Mark a nonce as seen.
122    fn mark_seen(&mut self, nonce: u64) {
123        if nonce > self.highest {
124            return; // Will be marked after shift
125        }
126        let diff = self.highest - nonce;
127        if diff >= REPLAY_WINDOW_SIZE as u64 {
128            return; // Too old
129        }
130        let bit_index = diff as usize;
131        let word_index = bit_index / 64;
132        let bit_offset = bit_index % 64;
133        self.bitmap[word_index] |= 1 << bit_offset;
134    }
135
136    /// Shift the window forward.
137    ///
138    /// When we receive a new highest nonce, we need to shift all existing bits
139    /// to make room for the new highest at position 0.
140    /// Bit position represents (highest - nonce), so older nonces have higher bit positions.
141    fn shift_window(&mut self, shift: u64) {
142        if shift >= REPLAY_WINDOW_SIZE as u64 {
143            // Complete reset - all previous nonces fall outside the window
144            self.bitmap = [0; REPLAY_WINDOW_SIZE / 64];
145            return;
146        }
147
148        let shift_words = (shift / 64) as usize;
149        let shift_bits = (shift % 64) as u32;
150
151        // Shift whole words (towards higher indices = older nonces)
152        if shift_words > 0 {
153            // Shift from high to low to avoid overwriting
154            for i in (shift_words..self.bitmap.len()).rev() {
155                self.bitmap[i] = self.bitmap[i - shift_words];
156            }
157            // Clear the newly freed low words
158            for word in self.bitmap.iter_mut().take(shift_words) {
159                *word = 0;
160            }
161        }
162
163        // Shift remaining bits within words (towards higher bit positions)
164        if shift_bits > 0 {
165            let mut carry = 0u64;
166            // Process from highest index to lowest (shift bits up within each word)
167            for i in (0..self.bitmap.len()).rev() {
168                let new_carry = self.bitmap[i] >> (64 - shift_bits);
169                self.bitmap[i] = (self.bitmap[i] << shift_bits) | carry;
170                carry = new_carry;
171            }
172        }
173    }
174
175    /// Reset the window (e.g., after rekey).
176    pub fn reset(&mut self) {
177        self.bitmap = [0; REPLAY_WINDOW_SIZE / 64];
178        self.highest = 0;
179        self.initialized = false;
180    }
181}

184    fn default() -> Self {
185        Self::new()
186    }
187}

189/// A complete crypto session for secure communication.
190///
191/// Combines key management, nonce construction, AEAD, and anti-replay
192/// into a single interface.
193pub struct CryptoSession {
194    /// Session ID
195    session_id: SessionId,
196    /// Our role (initiator or responder)
197    role: Role,
198    /// Current send key
199    send_key: SessionKey,
200    /// Current receive key
201    recv_key: SessionKey,
202    /// Rekey state (epoch, counters)
203    rekey_state: RekeyState,
204    /// Replay window for incoming packets
205    replay_window: ReplayWindow,
206    /// Old key retention for late packets during rekey
207    old_keys: OldKeyRetention,
208    /// Handshake hash for key derivation
209    handshake_hash: [u8; HASH_SIZE],
210}
211
212impl CryptoSession {
213    /// Create a new crypto session after handshake completion.
214    pub fn new(
215        session_id: SessionId,
216        role: Role,
217        send_key: SessionKey,
218        recv_key: SessionKey,
219        handshake_hash: [u8; HASH_SIZE],
220    ) -> Self {
221        Self {
222            session_id,
223            role,
224            send_key,
225            recv_key,
226            rekey_state: RekeyState::new(),
227            replay_window: ReplayWindow::new(),
228            old_keys: OldKeyRetention::new(),
229            handshake_hash,
230        }
231    }
232
233    /// Get the session ID.
234    pub fn session_id(&self) -> &SessionId {
235        &self.session_id
236    }
237
238    /// Get the current role.
239    pub fn role(&self) -> Role {
240        self.role
241    }
242
243    /// Get the current epoch.
244    pub fn epoch(&self) -> u32 {
245        self.rekey_state.epoch()
246    }
247
248    /// Check if we should initiate a rekey.
249    pub fn should_rekey(&self) -> bool {
250        self.rekey_state.should_rekey()
251    }
252
253    /// Check if keys are expired (session must terminate).
254    pub fn keys_expired(&self) -> bool {
255        self.rekey_state.keys_expired()
256    }
257
258    /// Get the direction for sending based on our role.
259    fn send_direction(&self) -> Direction {
260        match self.role {
261            Role::Initiator => Direction::InitiatorToResponder,
262            Role::Responder => Direction::ResponderToInitiator,
263        }
264    }
265
266    /// Get the direction for receiving based on our role.
267    fn recv_direction(&self) -> Direction {
268        self.send_direction().opposite()
269    }
270
271    /// Encrypt a frame for sending.
272    ///
273    /// Returns (nonce_counter, ciphertext).
274    pub fn encrypt_frame(
275        &mut self,
276        frame_type: u8,
277        flags: u8,
278        plaintext: &[u8],
279    ) -> Result<(u64, Vec<u8>), CryptoError> {
280        // Get counter and construct nonce
281        let counter = self.rekey_state.increment_send()?;
282        let nonce = construct_nonce(self.rekey_state.epoch(), self.send_direction(), counter);
283
284        // Construct AAD
285        let aad = construct_aad(frame_type, flags, self.session_id.as_bytes(), counter);
286
287        // Encrypt
288        let ciphertext = encrypt(&self.send_key, &nonce, &aad, plaintext)?;
289
290        Ok((counter, ciphertext))
291    }
292
293    /// Decrypt a received frame.
294    ///
295    /// Performs replay check BEFORE decryption per spec.
296    pub fn decrypt_frame(
297        &mut self,
298        frame_type: u8,
299        flags: u8,
300        nonce_counter: u64,
301        ciphertext: &[u8],
302    ) -> Result<Vec<u8>, CryptoError> {
303        // 1. Replay check FIRST (cheap, prevents DoS)
304        if self.replay_window.is_replay(nonce_counter) {
305            return Err(CryptoError::ReplayDetected);
306        }
307
308        // Construct nonce and AAD
309        let nonce = construct_nonce(self.rekey_state.epoch(), self.recv_direction(), nonce_counter);
310        let aad = construct_aad(frame_type, flags, self.session_id.as_bytes(), nonce_counter);
311
312        // 2. Try current keys first
313        if let Ok(plaintext) = decrypt(&self.recv_key, &nonce, &aad, ciphertext) {
314            // 3. Update replay window only after successful verification
315            let _ = self.replay_window.check_and_update(nonce_counter);
316            self.rekey_state.record_recv(nonce_counter);
317            return Ok(plaintext);
318        }
319
320        // 4. Try old keys if within retention window
321        self.old_keys.clear_if_expired();
322        if let Some(old_recv_key) = self.get_old_recv_key() {
323            // Try with previous epoch's nonce
324            let old_epoch = self.rekey_state.epoch().saturating_sub(1);
325            let old_nonce = construct_nonce(old_epoch, self.recv_direction(), nonce_counter);
326
327            if let Ok(plaintext) = decrypt(old_recv_key, &old_nonce, &aad, ciphertext) {
328                // Note: Don't update replay window for old epoch packets
329                // (they have their own counter space)
330                return Ok(plaintext);
331            }
332        }
333
334        Err(CryptoError::DecryptionFailed)
335    }
336
337    /// Get the old receive key based on role.
338    fn get_old_recv_key(&self) -> Option<&SessionKey> {
339        match self.role {
340            Role::Initiator => self.old_keys.old_responder_key(),
341            Role::Responder => self.old_keys.old_initiator_key(),
342        }
343    }
344
345    /// Perform a rekey operation.
346    ///
347    /// Advances the epoch and derives new keys.
348    pub fn rekey(&mut self) -> Result<(), CryptoError> {
349        use super::rekey::derive_rekey_keys;
350
351        // Retain current keys
352        self.old_keys
353            .retain(self.send_key.clone(), self.recv_key.clone());
354
355        // Advance epoch
356        self.rekey_state.advance_epoch()?;
357
358        // Derive new keys
359        let (new_initiator_key, new_responder_key) =
360            derive_rekey_keys(&self.handshake_hash, self.rekey_state.epoch())?;
361
362        // Update keys based on role
363        match self.role {
364            Role::Initiator => {
365                self.send_key = new_initiator_key;
366                self.recv_key = new_responder_key;
367            }
368            Role::Responder => {
369                self.send_key = new_responder_key;
370                self.recv_key = new_initiator_key;
371            }
372        }
373
374        // Reset replay window for new epoch
375        self.replay_window.reset();
376
377        Ok(())
378    }

385    #[test]
386    fn test_replay_window_basic() {
387        let mut window = ReplayWindow::new();
388
389        // First packet should succeed
390        assert!(window.check_and_update(0).is_ok());
391
392        // Same packet should fail (replay)
393        assert!(window.check_and_update(0).is_err());
394
395        // New packet should succeed
396        assert!(window.check_and_update(1).is_ok());
397
398        // Out of order but in window should succeed
399        assert!(window.check_and_update(5).is_ok());
400        assert!(window.check_and_update(3).is_ok());
401        assert!(window.check_and_update(4).is_ok());
402        assert!(window.check_and_update(2).is_ok());
403
404        // All replays
405        assert!(window.check_and_update(0).is_err());
406        assert!(window.check_and_update(3).is_err());
407        assert!(window.check_and_update(5).is_err());
408    }

447    fn test_crypto_session_roundtrip() {
448        let session_id = SessionId::generate();
449        let send_key = SessionKey::from_bytes([0x01; 32]);
450        let recv_key = SessionKey::from_bytes([0x02; 32]);
451        let handshake_hash = [0x42; 32];
452
453        let mut initiator = CryptoSession::new(
454            session_id,
455            Role::Initiator,
456            send_key.clone(),
457            recv_key.clone(),
458            handshake_hash,
459        );
460
461        let mut responder = CryptoSession::new(
462            session_id,
463            Role::Responder,
464            recv_key.clone(),
465            send_key.clone(),
466            handshake_hash,
467        );
468
469        // Initiator sends
470        let plaintext = b"Hello, NOMAD!";
471        let (counter, ciphertext) = initiator.encrypt_frame(0x03, 0x00, plaintext).unwrap();
472
473        // Responder receives
474        let decrypted = responder
475            .decrypt_frame(0x03, 0x00, counter, &ciphertext)
476            .unwrap();
477        assert_eq!(decrypted, plaintext);
478
479        // Responder sends back
480        let reply = b"Hello back!";
481        let (reply_counter, reply_ciphertext) =
482            responder.encrypt_frame(0x03, 0x00, reply).unwrap();
483
484        // Initiator receives
485        let decrypted_reply = initiator
486            .decrypt_frame(0x03, 0x00, reply_counter, &reply_ciphertext)
487            .unwrap();
488        assert_eq!(decrypted_reply, reply);
489    }
